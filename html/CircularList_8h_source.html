517template <typename T>
518CircularList<T>::CircularList(const CircularList& other)
519 : head(nullptr), count(0) {
520 if (other.empty()) return;
521
522 try {
523 Node* current = other.head;
524 if (!current) throw std::runtime_error("Invalid source list head");
525
526 size_t elements_copied = 0;
527 do {
528 if (!current->next || !current->prev) {
529 throw std::runtime_error("Invalid node links in source list");
530 }
531 push_back(current->data);
532 current = current->next;
533 elements_copied++;
534
535 if (elements_copied > other.count) {
536 throw std::runtime_error("Circular list structure corrupted");
537 }
538 } while (current != other.head);
539
540 if (elements_copied != other.count) {
541 throw std::runtime_error("Element count mismatch during copy");
542 }
543 } catch (...) {
544 clear();
545 throw;
546 }
547}

651 if (empty()) return true;

680 size_t elements_compared = 0;

683 elements_compared < std::min(size(), other.size())) {

740template <typename T>
741typename CircularList<T>::iterator CircularList<T>::insert(iterator pos,
742 const T& value) {
743 if (pos.node == nullptr || head == nullptr) {
744 push_back(value);
745 return iterator(head->prev, head);
746 }
